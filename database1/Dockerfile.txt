# Build Stage (Linux based .NET 7 SDK)
FROM mcr.microsoft.com/dotnet/sdk:7.0 AS build
WORKDIR /src

# Project file copy karein
COPY ["database1.csproj", "."]

# NuGet packages restore karein (Linux environment mein)
RUN dotnet restore "./database1.csproj"

# Baaki saara code copy karein
COPY . .

# Publish karein aur Windows paths ko block karein
RUN dotnet publish "database1.csproj" -c Release -o /app/out /p:RestoreFallbackFolders=""

# Runtime Stage (Linux based .NET 8 Runtime)
FROM mcr.microsoft.com/dotnet/aspnet:8.0
WORKDIR /app
COPY --from=build /app/out .

# Port settings
ENV ASPNETCORE_URLS=http://+:8080
EXPOSE 8080

ENTRYPOINT ["dotnet", "database1.dll"]
